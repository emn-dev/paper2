          if (Base.hasNamed(args, 'radius')) {
            center = Point.readNamed(args, 'center');
            radius = Size.readNamed(args, 'radius');

            var rect = Rectangle.readNamed(args, 'rectangle');

            radius = rect.getSize(true).divide(2);
